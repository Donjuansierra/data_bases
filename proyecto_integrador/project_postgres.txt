CREATE TABLE CUSTOMERS (
id SERIAL PRIMARY KEY,
name VARCHAR(256),
phone VARCHAR(10)
);

CREATE TABLE BRANCHES (
id INTEGER PRIMARY KEY NOT NULL,
name TEXT,
address TEXT
);

CREATE TABLE ORDERS (
id INTEGER PRIMARY KEY NOT NULL,
customer_id INTEGER,
branch_id INTEGER,
order_date NUMERIC,
total REAL
);

CREATE TABLE STOCKS (
id INTEGER PRIMARY KEY NOT NULL,
branch_id INTEGER NOT NULL,
product_id INTEGER NOT NULL,
quantity INTEGER,
UNIQUE (branch_id, product_id)
);

CREATE TABLE ITEMS (
id INTEGER PRIMARY KEY NOT NULL,
order_id INTEGER,
product_id INTEGER,
quantity INTEGER,
sale_amount REAL
);

CREATE TABLE PRODUCTS (
id INTEGER PRIMARY KEY NOT NULL,
name TEXT,
brand TEXT,
category_id INTEGER,
unit_price REAL
);

CREATE TABLE CATEGORIES (
id INTEGER PRIMARY KEY NOT NULL,
name TEXT
);
